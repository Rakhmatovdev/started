#include <iostream>
using namespace std;

int countA(string str) {
    int count = 0;
    for (int i = 0; i < str.length(); ++i) {
        if (str[i] == 'A')count++;

    }
    return count;

}

int main() {

    string str;
    cout << "Satr kiriting: ";
    getline(cin, str);
    cout << "Natija: " << countA(str);
}
